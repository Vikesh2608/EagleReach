<script> // ------------------------------ // Backend base // ------------------------------ const API_BASE = (window.location.hostname.includes('github.io')) ? 'https://eaglereach-1.onrender.com' : 'http://localhost:8000'; // Small helpers const $ = (id) => document.getElementById(id); const esc = (s) => String(s ?? '').replace(/[&<>"']/g, (m) => ({'&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;',"'":'&#39;'}[m])); const delay = (ms) => new Promise(r => setTimeout(r, ms)); function itemHTML(p) { if (!p) return ''; const photo = p.photo ? `<img src="${esc(p.photo)}" alt="">` : `<img src="https://via.placeholder.com/48" alt="">`; const name = esc(p.name || 'Unknown'); const role = esc([p.role, p.party, p.state && p.state.toUpperCase(), p.district && `#${p.district}`] .filter(Boolean).join(' • ')); const website = p.website ? `<a href="${esc(p.website)}" target="_blank" rel="noopener">official website</a>` : ''; const phone = p.phone ? `<span> | ${esc(p.phone)}</span>` : ''; const twitter = p.twitter ? ` <a href="https://twitter.com/${esc(p.twitter)}" target="_blank" rel="noopener">@${esc(p.twitter)}</a>` : ''; return ` <div class="official"> ${photo} <div class="meta"> <div class="name">${name}</div> <div class="sub">${role}</div> <div class="sub">${website}${phone}${twitter}</div> </div> </div>`; } function renderOfficials(data) { const off = data.officials || {}; const senators = Array.isArray(off.senators) ? off.senators : []; const reps = Array.isArray(off.representatives) ? off.representatives : (off.representative ? [off.representative] : []); $('senators').innerHTML = senators.length ? senators.map(itemHTML).join('') : '<div class="empty">No data.</div>'; $('representatives').innerHTML = reps.length ? reps.map(itemHTML).join('') : '<div class="empty">No data.</div>'; $('mayor').innerHTML = off.mayor ? itemHTML(off.mayor) : '<div class="empty">No data.</div>'; } function renderLocation(loc) { $('locBadge').textContent = (loc?.city && loc?.state) ? `${loc.city}, ${loc.state}` : (loc?.zip ? `ZIP ${loc.zip}` : ''); } function showBanner(msg = '', type = 'error') { const el = $('apiWarning'); if (!msg) { el.style.display = 'none'; return; } el.textContent = msg; el.className = `banner ${type}`; el.style.display = 'block'; } function setLoading(isLoading) { $('searchBtn').disabled = isLoading; $('locBtn').disabled = isLoading; $('zip').disabled = isLoading; } async function apiGet(path) { const r = await fetch(`${API_BASE}${path}`); if (!r.ok) { const text = await r.text().catch(()=> ''); throw new Error(`HTTP ${r.status} ${r.statusText} — ${text}`.slice(0, 300)); } return r.json(); } // --- Lookups ---------------------------------------------------- async function lookupOfficialsByZip(zip) { setLoading(true); showBanner(`Looking up officials for ZIP ${zip}...`, 'info'); try { const data = await apiGet(`/officials?zip=${encodeURIComponent(zip)}`); showBanner(''); renderLocation(data.location); renderOfficials(data); } catch (e) { console.error(e); showBanner(`Couldn't load officials. ${e.message}`); } finally { setLoading(false); } } async function lookupOfficialsByGeo(lat, lon) { setLoading(true); showBanner(`Finding officials for your exact location…`, 'info'); try { const data = await apiGet(`/officials/by_geo?lat=${lat}&lon=${lon}`); showBanner(''); renderLocation(data.location); renderOfficials(data); } catch (e) { console.error(e); showBanner(`Couldn't load officials from your location. ${e.message}`); } finally { setLoading(false); } } // Optional: news buttons (kept simple) async function loadNews(city, state) { try { $('openLocalNews').onclick = () => window.open(`https://news.google.com/search?q=${encodeURIComponent((city||'') + ' ' + (state||''))}&hl=en-US&gl=US&ceid=US:en`, '_blank'); $('worldNews').onclick = () => window.open('https://news.google.com/topstories?hl=en- US&gl=US&ceid=US:en','_blank'); } catch(e) {} } // --- Warm-up: ping /health once so Render free tier wakes up ---- async function warmBackend() { try { showBanner('Waking the server (Render free tier)…', 'info'); await Promise.race([ apiGet('/health'), (async ()=> { await delay(8000); throw new Error('warmup timeout'); })() ]); showBanner(''); } catch { // even if warmup times out, we carry on; the next request will still wake it } } // --- Wire up UI ------------------------------------------------- window.addEventListener('DOMContentLoaded', async () => { warmBackend(); $('searchForm').addEventListener('submit', (ev) => { ev.preventDefault(); const zip = $('zip').value.trim(); if (/^\d{5}$/.test(zip)) lookupOfficialsByZip(zip); else showBanner('Please enter a valid 5-digit ZIP.'); }); $('locBtn').addEventListener('click', () => { showBanner('Requesting browser location…', 'info'); setLoading(true); if (!navigator.geolocation) { showBanner('Geolocation not supported in this browser. Type your ZIP instead.'); setLoading(false); return; } navigator.geolocation.getCurrentPosition(async (pos) => { const { latitude, longitude } = pos.coords || {}; if (typeof latitude !== 'number' || typeof longitude !== 'number') { showBanner('Location unavailable. Please enter your ZIP.'); setLoading(false); return; } await lookupOfficialsByGeo(latitude, longitude); }, (err) => { showBanner(`Location permission denied. Please enter your ZIP manually. (${err.message || 'geolocation error'})`); setLoading(false); }, { enableHighAccuracy: false, timeout: 15000, maximumAge: 300000 }); }); // Auto-search if a 5-digit value is already present const initial = $('zip').value.trim(); if (/^\d{5}$/.test(initial)) lookupOfficialsByZip(initial); }); </script>
